# Bootstrap S3 remote state bucket and DynamoDB locking table
cd cloud/w8/mon/vpc-ec2-rds-s3-lab/bootstrap-backend
terraform init
terraform apply

# Create backend.hcl from bootstrap output
cd ..
Copy-Item backend.hcl.example backend.hcl
# Edit backend.hcl bucket and dynamodb_table with bootstrap output values

# Deploy lab infrastructure
terraform init -backend-config=backend.hcl
terraform validate
terraform plan
terraform apply

# Test web server
terraform output web_url

# Destroy lab before destroying backend
terraform destroy
cd bootstrap-backend
terraform destroy
